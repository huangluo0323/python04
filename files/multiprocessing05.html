多任务

案例：多进程拷贝文件夹

#coding=utf-8
import multiprocessing
import os
import time
import random

def copy_file(q,file_name,source_folder_name,dest_folder_name):
    '''拷贝文件到指定路径'''
    #分别打开文件
    with open(source_folder_name+'/'+file_name,'rb') as fr,open(dest_folder_name+'/'+file_name,'wb') as fw:
        while True:
            time.sleep(random.random())#适当延迟
            content = fr.read(1024)#读取
            if content:
                fw.write(content) #写入
            else:
                break
    q.put(file_name)#将已经备份的文件名放入队列

def main():
    source_folder_name = input('请输入要拷贝的文件夹名字：')
    #生成新的文件夹名
    dest_folder_name = source_folder_name+'[备份]'
    try:
        os.mkdir(dest_folder_name) #创建文件夹
    except:
        pass
    file_name_list = os.listdir(source_folder_name) #获得文件夹中所有文件名
    #创建队列
    q = multiprocessing.Manager().Queue()

    #创建进程池
    pool = multiprocessing.Pool(3)

    #遍历所有文件夹中文件，分别进行任务copyfile
    for file_name in file_name_list:
        pool.apply_async(copy_file,args=(q,file_name,source_folder_name,dest_folder_name))

    #不再添加任务
    pool.close()
    #总文件数
    all_num = len(file_name_list)
    while 1:
        #已经完成的文件
        file_name = q.get()
        #如果已经完成，则从原列表中删除
        if file_name in file_name_list:
            file_name_list.remove(file_name)
            #进度
        copy_rate = (all_num-len(file_name_list))*100/all_num
        print(f'进度:{file_name}{copy_rate}%')
        #超过100%则退出
        if copy_rate >= 100:
            break
if __name__ == '__main__':
    main()
